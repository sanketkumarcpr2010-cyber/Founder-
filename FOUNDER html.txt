// ── FORMAT ─────────────────────────────────────────────────────────
function fmtSecs(s) {
  return [Math.floor(s/3600), Math.floor((s%3600)/60), s%60]
    .map(n => String(n).padStart(2,'0')).join(':');
}

function fmtDuration(s) {
  if(s < 60)   return `${s}s`;
  if(s < 3600) return `${Math.floor(s/60)}m ${s%60}s`;
  return `${Math.floor(s/3600)}h ${Math.floor((s%3600)/60)}m`;
}

function fmtDate(d) {
  if(!d) return '';
  return new Date(d + 'T00:00:00').toLocaleDateString('en-IN',{day:'numeric',month:'short'});
}

function fmtDateTime(ts) {
  if(!ts) return '';
  const d = new Date(ts);
  return d.toLocaleDateString('en-IN',{day:'numeric',month:'short',year:'2-digit'}) + ' ' +
         d.toLocaleTimeString('en-IN',{hour:'2-digit',minute:'2-digit'});
}

function nowISO() { return new Date().toISOString(); }

function getDueClass(v) {
  if(!v) return '';
  const t = new Date().toISOString().split('T')[0];
  if(v < t) return 'overdue';
  if(v === t) return 'today';
  return '';
}

// ── TIMER RING ─────────────────────────────────────────────────────
const R = 105;
const CIRC = +(2 * Math.PI * R).toFixed(2); // 659.73

function initRing() {
  ['ringProg','ringGlow'].forEach(id => {
    const el = document.getElementById(id);
    el.style.strokeDasharray  = CIRC;
    el.style.strokeDashoffset = CIRC;
  });
}

function updateRing(s) {
  const frac   = Math.min(s % 3600, 3600) / 3600;
  const offset = CIRC - frac * CIRC;
  document.getElementById('ringProg').style.strokeDashoffset = offset;
  document.getElementById('ringGlow').style.strokeDashoffset = offset;
}

function setRingColor(color) {
  document.getElementById('ringProg').style.stroke = color;
  document.getElementById('ringGlow').style.stroke = color;
}

// ── TIMER ──────────────────────────────────────────────────────────
function startTimer() {
  if(timerRunning) return;
  timerRunning = true;
  if(!sessionStart) sessionStart = new Date();
  document.getElementById('timerDisplay').className = 'timer-display running';
  document.getElementById('timerStatus').textContent = 'RUNNING';
  document.getElementById('timerLabel').textContent  = '● Recording session...';
  setRingColor('var(--accent3)');
  timerInterval = setInterval(() => {
    timerSeconds++;
    document.getElementById('timerDisplay').textContent = fmtSecs(timerSeconds);
    updateRing(timerSeconds);
  }, 1000);
}

function pauseTimer() {
  if(!timerRunning) return;
  clearInterval(timerInterval);
  timerRunning = false;
  document.getElementById('timerDisplay').className = 'timer-display paused';
  document.getElementById('timerStatus').textContent = 'PAUSED';
  document.getElementById('timerLabel').textContent  = '⏸ Paused — click Start to resume';
  setRingColor('var(--yellow)');
}

function stopTimer() {
  if(timerSeconds === 0) { alert('No session running. Start the timer first!'); return; }
  clearInterval(timerInterval);
  timerRunning = false;
  const label = document.getElementById('sessionName').value.trim() || 'Unnamed Session';
  timeLogs.unshift({
    id: Date.now(),
    label,
    duration: timerSeconds,
    startTime: sessionStart
      ? sessionStart.toLocaleTimeString('en-IN',{hour:'2-digit',minute:'2-digit'})
      : new Date().toLocaleTimeString('en-IN',{hour:'2-digit',minute:'2-digit'}),
    ts: nowISO()
  });
  save();
  renderLogs(); renderStats(); renderHistory();
  resetTimer();
  document.getElementById('sessionName').value = '';
  document.getElementById('timerLabel').textContent = '✓ Session saved to log!';
  setTimeout(() => { document.getElementById('timerLabel').textContent = 'Label your session below'; }, 2500);
}

function resetTimer() {
  clearInterval(timerInterval);
  timerRunning = false; timerSeconds = 0; sessionStart = null;
  document.getElementById('timerDisplay').textContent  = '00:00:00';
  document.getElementById('timerDisplay').className    = 'timer-display';
  document.getElementById('timerStatus').textContent   = 'READY';
  document.getElementById('timerLabel').textContent    = 'Label your session below';
  setRingColor('var(--accent)');
  initRing();
}

// ── TIME LOG ───────────────────────────────────────────────────────
function renderLogs() {
  const el = document.getElementById('logList');
  document.getElementById('logCount').textContent = timeLogs.length;
  if(!timeLogs.length) {
    el.innerHTML = '<div class="empty-state">No sessions logged yet.<br>Start the timer to begin.</div>';
    return;
  }
  el.innerHTML = timeLogs.map((l,i) => `
    <div class="log-item">
      <span class="sno">${i+1}</span>
      <span class="log-time">${l.startTime}</span>
      <span class="log-label">${l.label}</span>
      <span class="log-duration">${fmtDuration(l.duration)}</span>
      <span class="log-date">${fmtDateTime(l.ts)}</span>
      <button class="del-btn" onclick="deleteLog(${l.id})">✕</button>
    </div>`).join('');
}

function deleteLog(id) {
  timeLogs = timeLogs.filter(l => l.id !== id);
  save(); renderLogs(); renderStats(); renderHistory();
}

// ── TO-DO ──────────────────────────────────────────────────────────
function addTodo() {
  const txt = document.getElementById('todoInput').value.trim();
  if(!txt) return;
  todos.unshift({ id: Date.now(), text: txt,
    dueDate: document.getElementById('todoDate').value,
    done: false, doneAt: null, ts: nowISO() });
  document.getElementById('todoInput').value = '';
  document.getElementById('todoDate').value  = '';
  save(); renderTodos(); renderStats();
}

function toggleTodo(id) {
  const t = todos.find(t => t.id === id);
  if(!t) return;
  t.done   = !t.done;
  t.doneAt = t.done ? nowISO() : null;
  save(); renderTodos(); renderStats(); renderHistory();
}

function deleteTodo(id) {
  todos = todos.filter(t => t.id !== id);
  save(); renderTodos(); renderStats(); renderHistory();
}

function renderTodos() {
  const el = document.getElementById('todoList');
  document.getElementById('todoCount').textContent = todos.length;
  if(!todos.length) { el.innerHTML = '<div class="empty-state">No tasks yet. Add one above!</div>'; return; }
  el.innerHTML = todos.map((t,i) => {
    const dc = getDueClass(t.dueDate);
    return `
    <div class="todo-item ${t.done ? 'done' : ''}">
      <span class="sno" style="${t.done ? 'opacity:0.4' : ''}">${i+1}</span>
      <div class="todo-check ${t.done ? 'checked' : ''}" onclick="toggleTodo(${t.id})"></div>
      <span class="todo-text">${t.text}</span>
      ${t.dueDate
        ? `<span class="todo-due ${dc}">${fmtDate(t.dueDate)}${dc==='overdue' ? ' ⚠' : dc==='today' ? ' ★' : ''}</span>`
        : `<span style="min-width:65px;flex-shrink:0"></span>`}
      <span class="ts-small">${fmtDateTime(t.ts)}</span>
      <button class="del-btn" onclick="deleteTodo(${t.id})">✕</button>
    </div>`;
  }).join('');
}

// ── BACKLOG ────────────────────────────────────────────────────────
function addBacklog() {
  const txt = document.getElementById('backlogInput').value.trim();
  if(!txt) return;
  backlog.unshift({ id: Date.now(), text: txt,
    priority:   document.getElementById('backlogPriority').value,
    targetDate: document.getElementById('backlogDate').value,
    ts: nowISO() });
  document.getElementById('backlogInput').value = '';
  document.getElementById('backlogDate').value  = '';
  save(); renderBacklog(); renderStats(); renderHistory();
}

function deleteBacklog(id) {
  backlog = backlog.filter(b => b.id !== id);
  save(); renderBacklog(); renderStats(); renderHistory();
}

function renderBacklog() {
  const el = document.getElementById('backlogList');
  document.getElementById('backlogCount').textContent = backlog.length;
  if(!backlog.length) { el.innerHTML = '<div class="empty-state">Backlog is clean!</div>'; return; }
  const order = {high:0, med:1, low:2};
  const sorted = [...backlog].sort((a,b) => order[a.priority] - order[b.priority]);
  el.innerHTML = sorted.map((b,i) => `
    <div class="backlog-item">
      <span class="sno">${i+1}</span>
      <span class="priority-badge p-${b.priority}">${b.priority.toUpperCase()}</span>
      <span class="backlog-text">${b.text}</span>
      <span class="ts-small" style="min-width:65px">${b.targetDate ? fmtDate(b.targetDate) : '—'}</span>
      <span class="ts-small">${fmtDateTime(b.ts)}</span>
      <button class="del-btn" onclick="deleteBacklog(${b.id})">✕</button>
    </div>`).join('');
}

// ── STATS ──────────────────────────────────────────────────────────
function renderStats() {
  const secs = timeLogs.reduce((a,l) => a + l.duration, 0);
  document.getElementById('statSessions').textContent = timeLogs.length;
  document.getElementById('statHours').textContent    = (secs/3600).toFixed(1) + 'h';
  document.getElementById('statDone').textContent     = todos.filter(t => t.done).length;
  document.getElementById('statBacklog').textContent  = backlog.length;
}

// ── HISTORY TABS ───────────────────────────────────────────────────
function switchTab(btn, tab) {
  currentTab = tab;
  document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
  btn.classList.add('active');
  renderHistory();
}

function renderHistory() {
  const el = document.getElementById('historyList');

  if(currentTab === 'sessions') {
    if(!timeLogs.length) { el.innerHTML = '<div class="empty-state">No sessions yet.</div>'; return; }
    el.innerHTML = timeLogs.map((l,i) => `
      <div class="log-item">
        <span class="sno">${i+1}</span>
        <span class="log-time">${l.startTime}</span>
        <span class="log-label">${l.label}</span>
        <span class="log-duration">${fmtDuration(l.duration)}</span>
        <span class="log-date">${fmtDateTime(l.ts)}</span>
      </div>`).join('');

  } else if(currentTab === 'tasks') {
    const done = todos.filter(t => t.done);
    if(!done.length) { el.innerHTML = '<div class="empty-state">No completed tasks yet.</div>'; return; }
    el.innerHTML = done.map((t,i) => `
      <div class="todo-item done">
        <span class="sno" style="opacity:0.4">${i+1}</span>
        <div class="todo-check checked"></div>
        <span class="todo-text">${t.text}</span>
        ${t.dueDate ? `<span class="todo-due">${fmtDate(t.dueDate)}</span>` : ''}
        <span class="ts-small">Added: ${fmtDateTime(t.ts)}</span>
        ${t.doneAt ? `<span class="ts-small" style="color:var(--accent3)">✓ Done: ${fmtDateTime(t.doneAt)}</span>` : ''}
      </div>`).join('');

  } else {
    const sorted = [...backlog].sort((a,b) => new Date(b.ts) - new Date(a.ts));
    if(!sorted.length) { el.innerHTML = '<div class="empty-state">No backlog history yet.</div>'; return; }
    el.innerHTML = sorted.map((b,i) => `
      <div class="backlog-item">
        <span class="sno">${i+1}</span>
        <span class="priority-badge p-${b.priority}">${b.priority.toUpperCase()}</span>
        <span class="backlog-text">${b.text}</span>
        <span class="ts-small" style="min-width:65px">${b.targetDate ? fmtDate(b.targetDate) : '—'}</span>
        <span class="ts-small">${fmtDateTime(b.ts)}</span>
      </div>`).join('');
  }
}
